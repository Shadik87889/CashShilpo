      const STORAGE_KEY = "cashshilpo_lang_pref";
      let currentLang = localStorage.getItem(STORAGE_KEY) || "bn";

      function translatePage(targetLang) {
        document.documentElement.lang = targetLang; // SEO Enhancement
        const body = document.body;

        if (targetLang === "bn") {
          body.classList.add("lang-bn");
          walkAndTranslate(body, "toBangla");
        } else {
          body.classList.remove("lang-bn");
          walkAndTranslate(body, "toEnglish");
        }

        const btnText = document.getElementById("lang-toggle-text");
        if (btnText) btnText.innerHTML = targetLang === "en" ? "বাং" : "EN";

        localStorage.setItem(STORAGE_KEY, targetLang);
        currentLang = targetLang;
      }

      function walkAndTranslate(node, direction) {
        // Translation ignores scripts, styles, native dual-language blocks, and icons
        if (
          ["SCRIPT", "STYLE", "NOSCRIPT"].includes(node.tagName) ||
          node.id === "lang-toggle-btn" ||
          (node.classList && node.classList.contains("lucide")) ||
          (node.classList && (node.classList.contains("bilingual-en") || node.classList.contains("bilingual-bn")))
        ) {
          return;
        }

        // Handle Text Nodes
        if (node.nodeType === 3) {
          const rawText = node.nodeValue;
          const cleanKey = rawText.replace(/\s+/g, " ").trim();

          if (!cleanKey) return;

          if (direction === "toBangla") {
            if (DICTIONARY[cleanKey]) {
              if (!node._originalText) node._originalText = rawText;
              node.nodeValue = DICTIONARY[cleanKey];
            }
          } else {
            if (node._originalText) {
              node.nodeValue = node._originalText;
              delete node._originalText;
            }
          }
          return;
        }

        // Handle Inputs & Textareas Placeholders
        if (node.tagName === "INPUT" || node.tagName === "TEXTAREA") {
          const val = node.getAttribute("placeholder");
          if (val) {
            const cleanVal = val.trim();
            if (direction === "toBangla") {
              if (DICTIONARY[cleanVal]) {
                if (!node._original_placeholder) node._original_placeholder = val;
                node.setAttribute("placeholder", DICTIONARY[cleanVal]);
              }
            } else {
              if (node._original_placeholder) {
                node.setAttribute("placeholder", node._original_placeholder);
                delete node._original_placeholder;
              }
            }
          }
        }

        node.childNodes.forEach((child) => walkAndTranslate(child, direction));
      }

          entries.forEach((entry) => {
            if (entry.isIntersecting) {
              entry.target.classList.add("is-visible");
              observer.unobserve(entry.target);
            }
          });
        }, generalObserverOptions);

        animatedElements.forEach((item) => {
          generalObserver.observe(item);
        });
