/**
 * CashShilpo Premium Invoice Templates Library
 * v2.1 - Fixes ReferenceError & Dynamic Data
 * * Features:
 * - 100% Light/Print-Friendly Themes
 * - Full parity with 'Futuristic' logic (EMI, Returns, AI Recs, Warranty)
 * - Dynamic Shop Details & Logos
 */

(function() {
    console.log("Initializing Premium Light-Theme Templates...");

    // --- Helpers ---
    const formatCurrency = (amount, currencyCode = 'USD') => {
        if (typeof window.currencyUtils !== 'undefined' && window.currencyUtils.format) {
            return window.currencyUtils.format(amount, currencyCode);
        }
        return new Intl.NumberFormat('en-US', { style: 'currency', currency: currencyCode }).format(amount);
    };

    const formatDate = (dateString) => {
        if (!dateString) return '';
        return new Date(dateString).toLocaleDateString(undefined, { year: 'numeric', month: 'short', day: 'numeric', hour: '2-digit', minute: '2-digit' });
    };

    // --- CSS Print Overrides for High Fidelity ---
    const printStyleFix = document.createElement('style');
    printStyleFix.innerHTML = `
        @media print {
            @page { margin: 0; size: auto; }
            body { -webkit-print-color-adjust: exact !important; print-color-adjust: exact !important; background-color: white !important; }
            #receipt-content { width: 100% !important; margin: 0 !important; box-shadow: none !important; border: none !important; }
            /* Force background colors to print */
            *[class*="bg-"] { -webkit-print-color-adjust: exact !important; print-color-adjust: exact !important; }
        }
    `;
    document.head.appendChild(printStyleFix);

    // Extracts common detailed data for all templates to use

        const { storeInfo, products } = window.firestoreData || { storeInfo: {}, products: [] };
        const { appState } = window;
        const taxRate = (appState?.settings?.taxRate ?? 0);
        const currency = invoice.currency || appState?.currentCurrency || 'USD';
        
        // Return Logic
        const isReturn = invoice.invoiceType === 'return';
        const title = isReturn ? 'CREDIT NOTE' : (invoice.isEMI ? 'EMI INVOICE' : 'TAX INVOICE');
        const totalLabel = isReturn ? 'Credited Amount' : 'Grand Total';
        
        // Financials
        const subtotal = invoice.subtotalInBaseCurrency;
        const tax = invoice.taxInBaseCurrency;
        const total = invoice.totalInBaseCurrency;
        const due = invoice.dueAmount || 0;
        
        // Discounts - FIX APPLIED HERE
        const discountDetails = invoice.persistentDiscountDetails || {};
        const manualDiscount = invoice.manualDiscountInBase || 0;
        
        let discountLabel = 'Discount';
        let discountAmount = discountDetails.amount || 0; // Fixed variable name from persistentDiscount to discountDetails
        
        if (discountAmount > 0 && discountDetails.best) {
             if (discountDetails.best === discountDetails.staff) discountLabel = `Staff Disc (${discountDetails.staff}%)`;
             else if (discountDetails.best === discountDetails.tier) discountLabel = `Tier Disc (${discountDetails.tier}%)`;
        }

        // Payments
        const payments = paymentState ? paymentState.payments : invoice.paymentDetails || [];
        const paidAmount = payments.reduce((sum, p) => sum + p.amount, 0);
        const changeDue = paymentState ? paidAmount - paymentState.total : 0;

        let recommendations = '';

        if (recs.length) recommendations = recs[0].name;

        // EMI
        let emiData = null;
        if (invoice.isEMI && window.firestoreData.emiSchedules) {
            emiData = window.firestoreData.emiSchedules.find(s => s.invoiceId === invoice.id);
        }

        return {
            store: storeInfo,
            currency,
            isReturn,
            title,
            totalLabel,
            subtotal,
            tax,
            total,
            due,
            discountLabel,
            discountAmount,
            manualDiscount,
            payments,
            paidAmount,
            changeDue,
            recommendations,
            emiData,
